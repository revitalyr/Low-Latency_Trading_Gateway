cmake_minimum_required(VERSION 3.20)
project(trading_gateway_tests)

# Find required packages
find_package(Threads REQUIRED)

# Tests CMakeLists.txt

# Use FetchContent for Boost.UT
include(FetchContent)
FetchContent_Declare(
    ut
    GIT_REPOSITORY https://github.com/boost-ext/ut
    GIT_TAG        v1.1.9
)
FetchContent_MakeAvailable(ut)

# Collect test files
set(TEST_SOURCES
    "test_main.cpp"
    "test_lockfree_queue.cpp"
    "test_message_parser.cpp"
    "test_matching_engine.cpp"
    "test_performance.cpp"
    "test_tcp_server.cpp"
    "test_gateway_server.cpp"
)

# Create test executable
add_executable(trading_gateway_tests ${TEST_SOURCES})

# Link libraries
target_link_libraries(trading_gateway_tests
    PRIVATE
    ut
    Threads::Threads
)

# Include directories
target_include_directories(trading_gateway_tests PRIVATE 
    ${CMAKE_SOURCE_DIR}/cpp-core
    ${CMAKE_SOURCE_DIR}/cpp-core/lockfree_queue
    ${CMAKE_SOURCE_DIR}/cpp-core/network
    ${CMAKE_SOURCE_DIR}/cpp-core/matching_engine
    ${CMAKE_SOURCE_DIR}/cpp-core/order_gateway
)

# Link our libraries if they exist
if(TARGET lockfree_queue)
    target_link_libraries(trading_gateway_tests PRIVATE lockfree_queue)
endif()
if(TARGET network)
    target_link_libraries(trading_gateway_tests PRIVATE network)
endif()
if(TARGET matching_engine)
    target_link_libraries(trading_gateway_tests PRIVATE matching_engine)
endif()

# Add tests
add_test(NAME AllTests COMMAND trading_gateway_tests)

# Set test properties
set_tests_properties(AllTests PROPERTIES TIMEOUT 120)
